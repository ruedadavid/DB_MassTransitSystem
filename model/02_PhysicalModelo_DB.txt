CREATE TABLE trip      (id serial, id_track int  , id_passenger int , d_start date          , created_by varchar(20) NOT NULL DEFAULT current_user, created_at date DEFAULT current_timestamp, updated_by varchar(20), updated_at date);
CREATE TABLE track     (id serial, id_station int, id_train int     , name varchar          , created_by varchar(20) NOT NULL DEFAULT current_user, created_at date DEFAULT current_timestamp, updated_by varchar(20), updated_at date);
